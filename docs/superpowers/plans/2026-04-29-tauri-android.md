/**
 * Typed shim over @mnlphlp/plugin-blec. The rest of the package only talks to
 * this interface so plugin-blec can be mocked in tests and swapped out later.
 */

export interface ScanResultDevice {
  address: string;
  name: string | null;
  isConnected: boolean;
  services: string[];
  serviceData: Record<string, number[]>;
  manufacturerData: Record<number, number[]>;
}

export type WriteType = 'withResponse' | 'withoutResponse';

export interface PluginBlecApi {
  checkPermissions: () => Promise<boolean>;
  requestPermissions: () => Promise<boolean>;
  startScan: (timeoutMs: number) => Promise<void>;
  stopScan: () => Promise<void>;
  getScanningUpdates: (handler: (devices: ScanResultDevice[]) => void) => Promise<() => void>;
  connect: (address: string, onDisconnect: () => void) => Promise<void>;
  disconnect: () => Promise<void>;
  send: (characteristic: string, data: Uint8Array, writeType: WriteType) => Promise<void>;
  subscribe: (characteristic: string, handler: (data: Uint8Array) => void) => Promise<() => void>;
  unsubscribe: (characteristic: string) => Promise<void>;
}

let injected: PluginBlecApi | undefined;

export function __setPluginBlecForTests(api: PluginBlecApi | undefined): void {
  injected = api;
}

export async function resolvePluginBlec(): Promise<PluginBlecApi> {
  if (injected) return injected;
  try {
    const mod = await import('@mnlphlp/plugin-blec');
    return mapModule(mod);
  } catch (cause) {
    throw new Error(
      '@mnlphlp/plugin-blec is unavailable; this build is not running inside a Tauri shell with the blec plugin registered.',
      { cause: cause as Error },
    );
  }
}

function mapModule(mod: typeof import('@mnlphlp/plugin-blec')): PluginBlecApi {
  return {
    checkPermissions: () => mod.checkPermissions(),
    requestPermissions: () => mod.requestPermissions(),
    startScan: (timeoutMs) => mod.startScan(timeoutMs),
    stopScan: () => mod.stopScan(),
    getScanningUpdates: (handler) =>
      mod.getScanningUpdates((devices) => handler(devices as unknown as ScanResultDevice[])),
    connect: (address, onDisconnect) => mod.connect(address, onDisconnect),
    disconnect: () => mod.disconnect(),
    send: (characteristic, data, writeType) => mod.send(characteristic, data, writeType),
    subscribe: (characteristic, handler) =>
      mod.subscribe(characteristic, (data: Uint8Array) => handler(data)),
    unsubscribe: (characteristic) => mod.unsubscribe(characteristic),
  };
}
```

> **Note:** Function names in `mapModule` follow the plugin-blec README at the time of writing. If a name has drifted (e.g. `sendString` vs `send`), adjust the call site here only — `PluginBlecApi` stays stable for our consumers. Run the typecheck after writing this file.

export interface TauriBlecAvailability {
  supported: boolean;
  reason?: string;
}

export function getTauriBlecAvailability(): TauriBlecAvailability {
  const win = (globalThis as { window?: { __TAURI_INTERNALS__?: unknown } }).window;
  if (!win) {
    return { supported: false, reason: '当前环境不是 Tauri 应用' };
  }
  if (!win.__TAURI_INTERNALS__) {
    return { supported: false, reason: '未检测到 Tauri 运行时（缺少 __TAURI_INTERNALS__）' };
  }
  return { supported: true };
}

import type { PluginBlecApi } from './plugin-blec.js';

/**
 * Minimal subset of BluetoothRemoteGATTCharacteristic that the Coyote protocol
 * adapter actually calls. Mirrors what `@dg-kit/protocol` accepts via its
 * `BluetoothRemoteGATTCharacteristicLike` interface.
 */
export interface PluginBlecCharacteristic extends EventTarget {
  readonly uuid: string;
  value: DataView | null;
  writeValue(data: Uint8Array): Promise<void>;
  writeValueWithoutResponse(data: Uint8Array): Promise<void>;
  startNotifications(): Promise<PluginBlecCharacteristic>;
  stopNotifications(): Promise<PluginBlecCharacteristic>;
}

export function createPluginBlecCharacteristic(
  uuid: string,
  api: PluginBlecApi,
): PluginBlecCharacteristic {
  const target = new EventTarget();
  let unsubscribe: (() => void) | null = null;
  let value: DataView | null = null;

  const ch: PluginBlecCharacteristic = Object.assign(target, {
    uuid,
    get value() {
      return value;
    },
    set value(next: DataView | null) {
      value = next;
    },
    async writeValue(data: Uint8Array): Promise<void> {
      await api.send(uuid, data, 'withResponse');
    },
    async writeValueWithoutResponse(data: Uint8Array): Promise<void> {
      await api.send(uuid, data, 'withoutResponse');
    },
    async startNotifications(): Promise<PluginBlecCharacteristic> {
      if (unsubscribe) return ch;
      unsubscribe = await api.subscribe(uuid, (bytes) => {
        value = new DataView(
          bytes.buffer.slice(bytes.byteOffset, bytes.byteOffset + bytes.byteLength),
        );
        target.dispatchEvent(new Event('characteristicvaluechanged'));
      });
      return ch;
    },
    async stopNotifications(): Promise<PluginBlecCharacteristic> {
      if (unsubscribe) {
        unsubscribe();
        unsubscribe = null;
      }
      await api.unsubscribe(uuid);
      return ch;
    },
  }) as unknown as PluginBlecCharacteristic;

  return ch;
}
```

```ts
import type { DeviceClient, DeviceCommand, DeviceCommandResult, DeviceState } from '@dg-kit/core';
import { resolvePluginBlec, type PluginBlecApi } from './plugin-blec.js';

export interface DiscoveredDevice {
  address: string;
  name: string | null;
  rssi: number | null;
  isConnected: boolean;
  services: string[];
}

export interface TauriBlecDeviceClientOptions {
  protocol: {
    subscribe(listener: (state: DeviceState) => void): () => void;
    onConnected(args: { deviceAddress: string; api: PluginBlecApi }): Promise<void>;
    onDisconnected(): Promise<void>;
    send(command: DeviceCommand): Promise<DeviceCommandResult>;
  };
  selectDevice: (devices: DiscoveredDevice[]) => Promise<string | null>;
  scanDurationMs?: number;
  /** Optional name prefix filter applied client-side after each scan update. */
  namePrefix?: string;
}

export class TauriBlecDeviceClient implements DeviceClient {
  private readonly listeners = new Set<(state: DeviceState) => void>();
  private connected = false;

  constructor(private readonly options: TauriBlecDeviceClientOptions) {
    options.protocol.subscribe((state) => this.emit(state));
  }

  async connect(): Promise<void> {
    const api = await resolvePluginBlec();

    const granted = await api.checkPermissions();
    if (!granted) {
      const ok = await api.requestPermissions();
      if (!ok) throw new Error('未授予蓝牙权限');
    }

    const scanDuration = this.options.scanDurationMs ?? 8000;
    const seen = new Map<string, DiscoveredDevice>();
    const stopUpdates = await api.getScanningUpdates((devices) => {
      for (const d of devices) {
        if (this.options.namePrefix && !(d.name ?? '').startsWith(this.options.namePrefix)) {
          continue;
        }
        seen.set(d.address, {
          address: d.address,
          name: d.name,
          rssi: null,
          isConnected: d.isConnected,
          services: d.services,
        });
      }
    });

    await api.startScan(scanDuration);

    let address: string | null = null;
    try {
      address = await this.options.selectDevice([...seen.values()]);
    } finally {
      stopUpdates();
      await api.stopScan().catch(() => undefined);
    }

    if (!address) {
      throw new Error('用户取消了设备选择');
    }

    await api.connect(address, () => {
      this.connected = false;
      void this.options.protocol.onDisconnected();
    });
    this.connected = true;
    await this.options.protocol.onConnected({ deviceAddress: address, api });
  }

  async disconnect(): Promise<void> {
    if (!this.connected) return;
    const api = await resolvePluginBlec();
    await api.disconnect().catch(() => undefined);
    this.connected = false;
    await this.options.protocol.onDisconnected();
  }

  async send(command: DeviceCommand): Promise<DeviceCommandResult> {
    return this.options.protocol.send(command);
  }

  subscribe(listener: (state: DeviceState) => void): () => void {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  }

  private emit(state: DeviceState): void {
    for (const l of this.listeners) l(state);
  }
}
```

> **Important architectural note for the implementer:**
>
> The Coyote `WebBluetoothProtocolAdapter` works on the abstract `BluetoothRemoteGATTCharacteristicLike` interface. plugin-blec exposes `send(uuid, bytes)` directly. There are two possible bridging strategies:
>
> 1. **Wrap each Coyote characteristic** as a `PluginBlecCharacteristic` (Task A4) and pass `{ device, server }`-shaped objects to `protocol.onConnected()` matching what `WebBluetoothProtocolAdapter` expects.
> 2. **Add a new method on the protocol adapter** like `onConnectedTauri({ deviceAddress, api })` that operates on plugin-blec's flat `(uuid, bytes)` API directly.
>
> The plan above assumes path #2 because it doesn't require synthesizing a `BluetoothRemoteGATTServer` mock. **Before writing this client, read `packages/protocol/src/web-bluetooth-protocol-adapter.ts`** (or wherever `WebBluetoothProtocolAdapter` lives) and verify which path is cheaper. If path #1 is feasible without changes to `@dg-kit/protocol`, take it — that means zero changes to the protocol package. Update the test and implementation accordingly.
>
> If path #2 is needed, that requires a small addition to `@dg-kit/protocol` (a new method) — add a `Task A4.5` here for it and write a corresponding changeset.

### Task C4: Implement DevicePicker

**Files:**

- Create: `apps/tauri-android/src/components/DevicePicker.tsx`
- Create: `apps/tauri-android/src/components/DevicePicker.module.css`
- Create: `apps/tauri-android/src/components/show-device-picker.ts`

The DevicePicker is a singleton modal that any code can open imperatively via `showDevicePicker(devices) → Promise<address|null>`. This matches the `selectDevice` callback signature `TauriBlecDeviceClient` expects.

- [ ] **Step 1: Implement `DevicePicker.tsx`**

```tsx
import React, { useEffect, useMemo, useState } from 'react';
import type { DiscoveredDevice } from '@dg-agent/device-tauri-ble';
import styles from './DevicePicker.module.css';

interface Props {
  open: boolean;
  devices: DiscoveredDevice[];
  onSelect: (address: string) => void;
  onCancel: () => void;
}

export function DevicePicker({ open, devices, onSelect, onCancel }: Props) {
  const sorted = useMemo(
    () => [...devices].sort((a, b) => (b.rssi ?? -999) - (a.rssi ?? -999)),
    [devices],
  );
  if (!open) return null;
  return (
    <div className={styles.backdrop} role="dialog" aria-modal="true">
      <div className={styles.panel}>
        <header className={styles.header}>选择郊狼设备</header>
        <ul className={styles.list}>
          {sorted.length === 0 ? (
            <li className={styles.empty}>正在扫描…若长时间未发现，请确认设备已开机</li>
          ) : (
            sorted.map((d) => (
              <li key={d.address}>
                <button className={styles.row} onClick={() => onSelect(d.address)}>
                  <span className={styles.name}>{d.name ?? '未知设备'}</span>
                  <span className={styles.address}>{d.address}</span>
                </button>
              </li>
            ))
          )}
        </ul>
        <footer className={styles.footer}>
          <button className={styles.cancel} onClick={onCancel}>
            取消
          </button>
        </footer>
      </div>
    </div>
  );
}
```

- [ ] **Step 3: Implement `show-device-picker.ts`**

```ts
import { createRoot, type Root } from 'react-dom/client';
import React from 'react';
import type { DiscoveredDevice } from '@dg-agent/device-tauri-ble';
import { DevicePicker } from './DevicePicker';

let host: HTMLDivElement | null = null;
let root: Root | null = null;

export function showDevicePicker(devices: DiscoveredDevice[]): Promise<string | null> {
  if (!host) {
    host = document.createElement('div');
    document.body.appendChild(host);
    root = createRoot(host);
  }
  return new Promise<string | null>((resolve) => {
    const close = (value: string | null) => {
      root?.render(React.createElement(React.Fragment));
      resolve(value);
    };
    root!.render(
      React.createElement(DevicePicker, {
        open: true,
        devices,
        onSelect: (address) => close(address),
        onCancel: () => close(null),
      }),
    );
  });
}
```

> **Important**: This is the simple version. `TauriBlecDeviceClient.connect()` calls `selectDevice(devices)` once with the device list it has accumulated at the time of the call. To get a _live-updating_ picker, in a follow-up iteration we'd refactor `selectDevice` to also subscribe to scan updates. For the first cut, the user sees whatever was found by the time scanning ends; this is acceptable.
